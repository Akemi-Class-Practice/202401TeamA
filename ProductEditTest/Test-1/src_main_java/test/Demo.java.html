package test;

public class Demo {
	public boolean isPrime(int n) {
		if (n < 2) return false;
		
		double rootN = Math.sqrt(n);
		
		for (int i = 2; i <= rootN; i++) {
			if (n % i == 0) {
				return false;
			}
		}

		return true;
	}
}
