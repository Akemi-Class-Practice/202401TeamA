package ecsite.com.controllers;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestParam;

import ecsite.com.services.AdminService;

@Controller
public class RegisterController {
	@Autowired
	private AdminService adminServuce;
	
	//登録画面の表示　/register
	@GetMapping("/register")
	public String getRegisterPage() {
		return "admin_register.html";
	}
	
	//管理者の登録処理　/register/process
	//もし、adminService.createAdmin == true　　admin_login.html
	//そうでない場合は、admin_register.html
	@PostMapping("/register/process")
	public String register(@RequestParam String adminName,
			@RequestParam String adminEmail,@RequestParam String password) {
		if(adminServuce.createAdmin(adminName,adminEmail,password)) {
			return "admin_login.html";
		}else {
			return "admin_register.html";
		}
	}

}
