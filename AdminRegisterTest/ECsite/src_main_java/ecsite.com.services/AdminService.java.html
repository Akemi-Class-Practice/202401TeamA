package ecsite.com.services;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import ecsite.com.models.AdminEntity;
import ecsite.com.repositories.AdminRepository;

@Service
public class AdminService {
		//管理者の登録処理と登録するためのチェック　boolean
		//もし、findByAdminEmail == null だったらsaveを使って保存処理をする⇒true
		//そうでない場合は、保存処理をしない=>false
	
		@Autowired
		private AdminRepository adminRepo;
		public boolean createAdmin(String name,String email,String password) {
			if(adminRepo.findByAdminEmail(email)==null) {
				adminRepo.save(new AdminEntity(email,name,password));
				return true;
			}else {
				return false;
			}
		}
		//管理者のログインのチェック　AdminEntity
		//もし、findByAdminEmailAndpassword == null　ログインしない　null
		//そうでない場合はログインをする ログインしている人の情報を渡す
		public AdminEntity checkLogin(String email, String password) {
			AdminEntity adminEntity = adminRepo.findByAdminEmailAndPassword(email, password);
			if(adminEntity == null) {
				return null;
			}else {
				return adminEntity;
			}
		}
		
}
